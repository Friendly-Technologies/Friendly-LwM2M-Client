77 bool Object::removeInstance(ID_T instanceId) {

86  if (NULL != element) delete element;

89  _instances.erase(inst);

116 ID_T Object::getFirstAvailableInstanceID() {
117  // Usually, each subsequent free index will be equal to the number of created objects
118  ID_T id = _instances.size();
119  if (id == ID_T_MAX_VAL) return id;
120  // But it won't always be like that
121  while (isInstanceExist(id) && id != ID_T_MAX_VAL) id++;
122  // It is also possible that all indexes after the current size are occupied
123  if (id == ID_T_MAX_VAL) {
124  id = 0;
125  // In this case, we need to check the indexes that are before the current size
126  while (isInstanceExist(id) && id < _instances.size()) id++;
127  }
128  return id;
