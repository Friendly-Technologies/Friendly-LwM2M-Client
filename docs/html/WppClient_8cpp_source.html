99 time_t WppClient::loop() {
100  // Max sleep time
101  time_t sleepTimeSec = _maxSleepTimeSec;
102 
103  WPP_LOGD(TAG_WPP_CLIENT, "Handling server packets if they exists");
104  // Handles packets retreived from server
105  if (connection().getPacketQueueSize()) connection().handlePacketsInQueue(*this);
106 
107  WPP_LOGD(TAG_WPP_CLIENT, "Handling wpp tasks if they exists");
108  // Handles Wpp tasks in the WppClient context and return next call interval
109  if (WppTaskQueue::getTaskCnt()) {
110  time_t nextTaskCallIntervalSec = WppTaskQueue::handleEachTask(*this);
111  if (nextTaskCallIntervalSec < sleepTimeSec) sleepTimeSec = nextTaskCallIntervalSec;
112  }
113 
114  // Handles wakaama core state
115  int result = lwm2m_step(_lwm2m_context, &sleepTimeSec);
116  WPP_LOGD(TAG_WPP_CLIENT, "Handling lwm2m internal state: result -> %d, state -> %d", result, getState());
117  if (result) {
118  WPP_LOGW(TAG_WPP_CLIENT, "LWM2M core step failed, error code: %d", result);
119  if (getState() == STATE_BOOTSTRAPPING || getState() == STATE_BOOTSTRAP_REQUIRED) {
120  WPP_LOGW(TAG_WPP_CLIENT, "Trying to restore security and server objects");
121  registry().object(OBJ_ID::LWM2M_SECURITY)->restore();
122  registry().object(OBJ_ID::LWM2M_SERVER)->restore();
123  }
124  _lwm2m_context->state = STATE_INITIAL;
125  }
126 
127  return sleepTimeSec;
128 }
