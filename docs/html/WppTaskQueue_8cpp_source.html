155 time_t WppTaskQueue::handleEachTask(WppClient& client) {
156  _handleTaskGuard.lock();
157 
158  _taskQueueGuard.lock();
159  // We create copy for be sure that adding new task not corrupted begin()
160  // and end() iterators inside _tasks list.
161  std::list<TaskInfo *> tasksCopy = _instance._tasks;
162  _taskQueueGuard.unlock();
163 
164  for (auto task : tasksCopy) {
165  // Be sure that we do not override SHOULD_BE_DELETED state
166  _taskQueueGuard.lock();
167  if (task->state & SHOULD_BE_DELETED || task->nextCallTime > WppPlatform::getTime()) {
168  _instance._taskQueueGuard.unlock();
169  continue;
170  }
171  task->state = EXECUTING;
172  _taskQueueGuard.unlock();
173 
174  // Here state can be changed to SHOULD_BE_DELETED but it is
175  // not matter becouse we have already set EXECUTING state look
176  // at description of requestToRemoveTask() and requestToRemoveEachTask()
177  bool isFinished = task->task(client, task->ctx);
178 
179  // Be sure that we do not override SHOULD_BE_DELETED state
180  _taskQueueGuard.lock();
181  if (isFinished || task->state & SHOULD_BE_DELETED) {
182  task->state = SHOULD_BE_DELETED;
183  } else {
184  task->state = IDLE;
185  task->nextCallTime = WppPlatform::getTime() + task->delaySec;
186  }
187  _taskQueueGuard.unlock();
188  }
189 
190  _instance.deleteFinishedTasks();
191  time_t nextCallInterval = _instance.updateNextCallTimeForTasks();
192  _handleTaskGuard.unlock();
193  return nextCallInterval;
194 }
